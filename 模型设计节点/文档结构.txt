huatai-agent/
├── main.py                  # 唯一入口：初始化主图、传入测试数据、处理人机交互(Human-in-the-loop)
├── requirements.txt         # 依赖包管理 (langgraph, langchain-openai, sqlalchemy 等)
│
├── agents/                  # 🧠 【指挥官】负责排兵布阵 (图的编排、状态定义、路由逻辑)
│   ├── state.py             # 核心：定义全局状态字典 (TypedDict)，贯穿三个 Agent 的数据流
│   ├── main_graph.py        # 主图：串联三个子图，处理 Security 到 Query 的跨图重试回退
│   ├── intent_agent.py      # 子图1：Intent Agent (处理模糊口径、业务指标检索与追问)
│   ├── query_agent.py       # 子图2：Query Agent (处理表结构检索、贴合度校验与 SQL 生成)
│   └── security_agent.py    # 子图3：Security Agent (处理危险拦截、权限注入、执行与反思)
│
├── nodes/                   # 👷 【打工人】负责单兵作战 (具体的业务逻辑函数，不包含连线)
│   ├── ambiguity_node.py    # 节点3：分析用户输入，判断是否有模糊口径和目标
│   ├── metric_search_node.py# 寻找业务口径数据库，匹配标准指标定义
│   ├── schema_node.py       # 节点4：根据意图检索相关的表结构 (DDL) 和列信息
│   ├── sql_gen_node.py      # 节点6：结合口径和表结构，调用 LLM 生成 SQL
│   ├── security_node.py     # 节点7 & 7.1：AST 语法树分析(拦截DROP)、数据权限校验与条件注入
│   ├── execution_node.py    # 节点8：安全执行 SQL 并捕获数据库返回的结果或报错
│   └── reflect_node.py      # 反思节点：分析 SQL 报错信息，生成修改建议 (为重试做准备)
│
└── tools/                   # 🛠️ 【装备库】负责连接外部世界 (与 LLM 无关的纯代码工具)
    ├── db_connector.py      # 数据库连接池配置、执行 SQL 的底层封装
    ├── ast_parser.py        # SQL 语法树解析工具 (用于精准识别 UPDATE/DELETE 等危险操作)
    ├── schema_retriever.py  # 从向量数据库 (如 FAISS/Chroma) 中检索表结构
    └── metric_retriever.py  # 从知识库中检索业务口径和指标定义
